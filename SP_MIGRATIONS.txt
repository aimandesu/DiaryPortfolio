ALTER PROCEDURE [dbo].[sp_GetMediaByUserId]
    @UserId  UNIQUEIDENTIFIER,
    @Skip    INT,
    @PerPage INT
AS
BEGIN
    SET NOCOUNT ON;

    -- Capture only the paged Media IDs first
    CREATE TABLE #PagedMediaIds (Id UNIQUEIDENTIFIER);

    INSERT INTO #PagedMediaIds (Id)
    SELECT m.Id
    FROM Medias m
    INNER JOIN Spaces s          ON m.SpaceId       = s.Id
    INNER JOIN DiaryPortfolios dp ON s.dp_id        = dp.Id
    WHERE dp.UserId = @UserId
    ORDER BY m.CreatedAt DESC
    OFFSET @Skip ROWS FETCH NEXT @PerPage ROWS ONLY;

    -- Result set 1: total count
    SELECT COUNT(*)
    FROM Medias m
    INNER JOIN Spaces s           ON m.SpaceId  = s.Id
    INNER JOIN DiaryPortfolios dp ON s.dp_id    = dp.Id
    WHERE dp.UserId = @UserId;

    -- Result set 2: paged media (flat row)
    SELECT
        m.Id,
        m.Title,
        m.Description,
        m.CreatedAt,
        ss.Selection  AS MediaStatus,
        ts.Selection  AS MediaType,
        sp.Id         AS Space_Id,
        sp.Title      AS Space_Title,
        l.Id          AS Loc_Id,
        l.Name        AS Loc_Name,
        l.Latitude    AS Loc_Latitude,
        l.Longitude   AS Loc_Longitude
    FROM Medias m
    INNER JOIN #PagedMediaIds pmi ON m.Id                    = pmi.Id
    LEFT JOIN  Selections ss      ON m.MediaStatusSelectionId = ss.Id
    LEFT JOIN  Selections ts      ON m.MediaTypeSelectionId   = ts.Id
    LEFT JOIN  Spaces sp          ON m.SpaceId                = sp.Id
    LEFT JOIN  Locations l        ON m.LocationId             = l.Id
    ORDER BY m.CreatedAt DESC;

    -- Result set 3: videos for THIS PAGE only
    SELECT v.Id, v.Url, v.Duration, v.Mime, v.Size, v.MediaId
    FROM Videos v
    WHERE v.MediaId IN (SELECT Id FROM #PagedMediaIds);

    -- Result set 4: photos for THIS PAGE only
    SELECT p.Id, p.Url, p.Mime, p.Width, p.Height, p.Size, p.MediaId
    FROM Photos p
    WHERE p.MediaId IN (SELECT Id FROM #PagedMediaIds);

    DROP TABLE #PagedMediaIds;
END

ALTER   PROCEDURE [dbo].[sp_UpsertPortfolioProfile]
    -- User
    @UserId UNIQUEIDENTIFIER,
    @UserName NVARCHAR(256),
    @NormalizedUserName NVARCHAR(256),
    @Email NVARCHAR(256),
    @NormalizedEmail NVARCHAR(256),

    -- Portfolio Profile
    @Name NVARCHAR(MAX),
    @Age INT = NULL,
    @Title NVARCHAR(MAX),
    @About NVARCHAR(MAX),
    @Address NVARCHAR(MAX),

    -- Location
    @LocationName NVARCHAR(MAX),
    @Latitude NVARCHAR(MAX) = '',
    @Longitude NVARCHAR(MAX) = '',

-- Photo (make nullable)
@PhotoUrl NVARCHAR(MAX) = NULL,
@PhotoMime NVARCHAR(MAX) = NULL,
@PhotoWidth FLOAT = NULL,
@PhotoHeight FLOAT = NULL,
@PhotoSize FLOAT = NULL,

-- File (Resume)
@FileUrl NVARCHAR(MAX) = NULL,
@FileDescription NVARCHAR(MAX) = NULL
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;

    BEGIN TRY

        -- 1. Update AspNetUsers
        UPDATE AspNetUsers
        SET 
            UserName = @UserName,
            NormalizedUserName = @NormalizedUserName,
            Email = @Email,
            NormalizedEmail = @NormalizedEmail
        WHERE Id = @UserId;

        -- 2. Get existing FK ids from PortfolioProfile
        DECLARE @ExistingLocationId UNIQUEIDENTIFIER;
        DECLARE @ExistingResumeId UNIQUEIDENTIFIER;
        DECLARE @ExistingPhotoId UNIQUEIDENTIFIER;
        DECLARE @ExistingFileId UNIQUEIDENTIFIER;

        SELECT 
            @ExistingLocationId = LocationId,
            @ExistingResumeId = ResumeId,
            @ExistingPhotoId = ProfilePhotoId
        FROM PortfolioProfile
        WHERE UserId = @UserId;

        -- Get existing FileId from Resume if Resume exists
        IF @ExistingResumeId IS NOT NULL
            SELECT @ExistingFileId = FileId FROM Resume WHERE Id = @ExistingResumeId;

        -- 3. Upsert Location
        DECLARE @LocationId UNIQUEIDENTIFIER = ISNULL(@ExistingLocationId, NEWID());
        IF @ExistingLocationId IS NULL
            INSERT INTO Locations (Id, Name, Latitude, Longitude)
            VALUES (@LocationId, @LocationName, @Latitude, @Longitude);
        ELSE
            UPDATE Locations SET Name = @LocationName WHERE Id = @LocationId;

        -- 4. Upsert Photo (always replace data, keep same row if exists)
        DECLARE @PhotoId UNIQUEIDENTIFIER = ISNULL(@ExistingPhotoId, NEWID());
        IF @ExistingPhotoId IS NULL
            INSERT INTO Photos (Id, Url, Mime, Width, Height, Size)
            VALUES (@PhotoId, @PhotoUrl, @PhotoMime, @PhotoWidth, @PhotoHeight, @PhotoSize);
        ELSE
            UPDATE Photos 
            SET Url = @PhotoUrl, Mime = @PhotoMime, Width = @PhotoWidth, 
                Height = @PhotoHeight, Size = @PhotoSize
            WHERE Id = @PhotoId;

        -- 5. Upsert File (resume file, always replace data)
        DECLARE @FileId UNIQUEIDENTIFIER = ISNULL(@ExistingFileId, NEWID());
        IF @ExistingFileId IS NULL
            INSERT INTO Files (Id, Url, Description)
            VALUES (@FileId, @FileUrl, @FileDescription);
        ELSE
            UPDATE Files 
            SET Url = @FileUrl, Description = @FileDescription
            WHERE Id = @FileId;

        -- 6. Upsert Resume
        DECLARE @ResumeId UNIQUEIDENTIFIER = ISNULL(@ExistingResumeId, NEWID());
        IF @ExistingResumeId IS NULL
            INSERT INTO Resume (Id, FileId)
            VALUES (@ResumeId, @FileId);
        ELSE
            UPDATE Resume SET FileId = @FileId WHERE Id = @ResumeId;

        -- 7. Update PortfolioProfile
        UPDATE PortfolioProfile
        SET
            Name = @Name,
            Age = @Age,
            Title = @Title,
            About = @About,
            Address = @Address,
            LocationId = @LocationId,
            ProfilePhotoId = @PhotoId,
            ResumeId = @ResumeId
        WHERE UserId = @UserId;

        COMMIT TRANSACTION;

    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END
